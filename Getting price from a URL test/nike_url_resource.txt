   * check the browser's userAgent and redirect it to the appropriate presentation
   */
  nike.exp.script.device_detect.redirectOnAgentMismatch = function(isPresentationMobile, alternateUrl){
    var newUrl,
    ref,
    loc = window.location,
    selectedHost,
    queryString,
    presentationMatchesDevice;

    if(isPresentationMobile === undefined || !alternateUrl){
      log("Presentation vars are not defined, so we are skipping the agent redirect check");

    } else {
      //Compare values as strings so we don't have to worry about object type
      presentationMatchesDevice = ("" + isPresentationMobile).toLowerCase() === ("" + nike.exp.script.device_detect.isMobile()).toLowerCase();
      log("isPresentationMobile=" + isPresentationMobile + " isMobileAgent=" + nike.exp.script.device_detect.isMobile() + " presentationMatchesDevice=" + presentationMatchesDevice);

      // If current presentation is a mismatch for the device and it was not manually selected as "dev", redirect user to the correct presentation
      // If the presentation is manually selected as "dev", do nothing here, unless presentation is set to "mobile" and the page is desktop, then switch.
      if ((!checkSPCookie("dev") && !presentationMatchesDevice && !checkSPCookie("mobile")) || (checkSPCookie("mobile") && isPresentationMobile == "false")){
        var newUrl = nike.exp.script.device_detect.buildNewUrl(alternateUrl);
        if (window.location != newUrl) {
          window.location = newUrl;
        }
      }
    }
  };

  /**
   * Combine query string params, making sure that the alternate url takes precedence.
   */
  nike.exp.script.device_detect.buildNewUrl = function(alternateUrl) {
    var self = nike.exp.script.device_detect;
    var newUrl = '';
    var oldUrl = alternateUrl;
    var oldParams = {};
    var newParams = {};
    var combinedParams = {};

    if(alternateUrl.indexOf('?') != -1) {
      var pieces = alternateUrl.split('?');
      oldUrl = pieces[0];
      var query = pieces[1];
      oldParams = self.buildParamMap(query);
    }

    var newQueryString = buildQueryString();
    // remove the question mark
    newQueryString = newQueryString.replace('?', '');
    newParams = self.buildParamMap(newQueryString);

    // combine the params
    var combinedParams = newParams;
    for(var key in oldParams) {
      combinedParams[key] = oldParams[key];
    }

    // build the query string
    var queryString = '';
    var tokens = [];
    for(var key in combinedParams) {
      var token = key + '=' + combinedParams[key];
      tokens.push(token);
    }

    if(tokens.length > 0) {
      queryString = '?' + tokens.join('&');
    } else {
      queryString = '';
    }
    newUrl = oldUrl + queryString;

    if(window.location.hash) {
      newUrl += window.location.hash;
    }

    return newUrl;
  };

  function getCookie(name) {

    var tempCookie = '';
    var cookieName = '';
    var cookieValue = '';
    var cookieFound = false; // set boolean t/f default f

      cookieName = '';
    }

    }
  }

    if(!/(^|&)ref=/.test(window.location.search)){

      }

      if(retQS && retQS.length > 0){
        retQS += "&ref=" + ref;
      } else{
        retQS = "?ref=" + ref;
      }
    }
    return retQS;
  }
